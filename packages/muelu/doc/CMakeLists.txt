# add a target to generate API documentation with Doxygen
FIND_PACKAGE(Doxygen)
IF(DOXYGEN_FOUND)
  CONFIGURE_FILE(${CMAKE_CURRENT_SOURCE_DIR}/Doxyfile.in ${CMAKE_CURRENT_BINARY_DIR}/Doxyfile @ONLY)
  ADD_CUSTOM_TARGET(
    doc_muelu
    TRILINOS_HOME=${PROJECT_SOURCE_DIR} ${DOXYGEN_EXECUTABLE} ${CMAKE_CURRENT_BINARY_DIR}/Doxyfile
    WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
    COMMENT "Generating ${PACKAGE_NAME} API documentation with Doxygen"
    VERBATIM
  )
ENDIF(DOXYGEN_FOUND)

IF(${PACKAGE_NAME}_ENABLE_Tutorial)
  TRIBITS_ADD_EXAMPLE_DIRECTORIES(Tutorial)
ELSE()
  MESSAGE(STATUS "disabling tutorial")
ENDIF()

ADD_SUBDIRECTORY(UsersGuide)
